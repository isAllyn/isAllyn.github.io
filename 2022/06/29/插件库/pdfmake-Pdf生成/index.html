pdfmake pdf自动生成

2022年06月29日
插件库
留言

install

"pdfmake": "^0.2.5",

yarn add pdfmake

use

const pdfmake = require("pdfmake/build/pdfmake");
const pdf_fonts = require("pdfmake/build/vfs_fonts");
pdfmake.vfs = pdf_fonts.pdfMake.vfs;
 var docDefinition = {
     content: "This is an sample PDF printed with pdfMake"
 };
 this.$pdfmake.createPdf(docDefinition).download("test.pdf");

example

[简书中文文档]https://www.jianshu.com/p/129554713e23

var docDefinition = { content: 'This is an sample PDF printed with pdfMake' };

pdfMake.createPdf(docDefinition).download(); // 下载

pdfMake.createPdf(docDefinition).open(); // 新窗口打开

pdfMake.createPdf(docDefinition).print(); // 打印

let docDefinition={
  content: [
    {
      layout: 'lightHorizontalLines', // optional
      table: {
        // headers are automatically repeated if the table spans over multiple pages
        // you can declare how many rows should be treated as headers
        headerRows: 1,
        widths: [ '*', 'auto', 100, '*' ],

        body: [ // body数组中每一个索引就是一行, 索引的值为一个数组, 数组中的每一个索引都是相对应的列,跟导出excel表格的数据结构一样
          [ 'First', 'Second', 'Third', 'The last one' ],
          [ 'Value 1', 'Value 2', 'Value 3', 'Value 4' ],
          [ { text: 'Bold value', bold: true }, 'Val 2', 'Val 3', 'Val 4' ]
        ]
      }
    }
  ]
}

页眉页脚

var docDefinition = {
    content: new Array(120).fill("asdasdasdasd"),
    header: function (currentPage, pageCount, pageSize) {
        // you can apply any logic and return any valid pdfmake element
        // currentPage:当前页 pageCount:页数 和 pageSize:页面大小
        return [
            { text: "simple text", alignment: currentPage % 2 ? "left" : "right" },
            { canvas: [{ type: "rect", x: 170, y: 32, w: pageSize.width - 170, h: 40 }] }
        ];
    }
};

背景层

var docDefinition = {
    content: new Array(120).fill("asdasdasdasd"),
    header: function (currentPage) {
        return [{ text: "simple text " + currentPage }];
    },
    // 设置背景
    background: function (currentPage, pageSize) {
        return `page ${currentPage} with size ${pageSize.width} x ${pageSize.height}`;
    }
};

边距

pdfMake 中的任何元素都可以有边距：

var docDefinition = {
    content: new Array(120).fill("asdasdasdasd"),
    header: function (currentPage) {
        return [{ text: "simple text " + currentPage }];
    },
    background: function (currentPage, pageSize) {
        return [
            {
                text: `page ${currentPage} with size ${pageSize.width} x ${pageSize.height}`,
                margin: [120, 100, 100, 100] // 设置一个边距
            }
        ];
    }
};

图片

这很简单。只需使用{ image: ‘…’ }节点类型。
支持 JPEG 和 PNG 格式。

  • dataUrl || pdfmake 提供的虚拟文件系统 || url 从图像中的名称引用加载图像
var docDefinition = {
    content: [
        // {
        //     // you'll most often use dataURI images on the browser side
        //     // if no width/height/fit is provided, the original size will be used
        //     image: "data:image/jpeg;base64,...encodedContent..." // 不配置宽高会使用原始宽高
        // },
        {
            // is supported loading images via url from reference by name in images
            image: "snow"
        },
        {
            image: "test"
        }
    ],
    background: function () {
        return [
            {
                image: "bg"
            }
        ];
    },
    images: {
        // is supported loading images via url (https or http protocol)
        snow: "https://picsum.photos/seed/picsum/500/700",
        test: "https://picsum.photos/seed/picsum/600/900",
        bg: {
            url: "https://picsum.photos/id/1080/367/267",
            headers: {
                myheader: "123",
                myotherheader: "abc"
            }
        },
        // is supported loading images via url with custom headers
        strawberries: {
            url: "https://picsum.photos/id/1080/367/267",
            headers: {
                myheader: "123",
                myotherheader: "abc"
            }
        }
    }
};

        modifying: true, //是否允许修改文件。指定true允许修改文档内容
        copying: false, //是否允许复制文本或图形。指定true允许复制
        annotating: true, //是否允许注释、填表。指定true允许注释和表单填写
        fillingForms: true, //是否允许填表和签名。指定true允许填写表单域和签名
        contentAccessibility: false, //是否允许复制可访问性文本。指定true以允许复制以实现可访问性
        documentAssembly: true //是否允许组装文件。指定true允许文档组装

git clone git@github.com:bpampuch/pdfmake.git

  1. npm install -g gulp

  2. 查看环境变量:npm config get prefix

  3. 在github，pafmake地址下载安装包pdfmake-master，打开这个工程，在package.json中devDependencies下添加一项
    https://github.com/bpampuch/pdfmake

    "gulp": "^3.9.1", // 跟全局版本最好一致

  4. npm install

  5. 找一个微软雅黑的ttf文件
    http://m.diyiziti.com/Download/537/

  6. 将pdfmake/fonts下腾空并放入微软雅黑.ttf文件

  7. https://gitee.com/talianshe/pafmake-fonts-build

  8. 生成vfs_fonts.js文件

  9. 文件改成export default 方式导出

  10. import vfs from "./vfs_fonts";
const pdfmake = require("pdfmake/build/pdfmake");

pdfmake.vfs = vfs;
pdfmake.fonts = {
    wryh: {
        normal: "微软雅黑.ttf",
        bold: "微软雅黑.ttf",
        italics: "微软雅黑.ttf",
        bolditalics: "微软雅黑.ttf"

};
