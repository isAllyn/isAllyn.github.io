<template>
  <div class="ButtonTem__">
    <div @click="btnEventClick" class="add-btn btn-position icon-theme"
         v-if=" typeStr==='add'">
      <i class="icon-i iconfontBtn icon-jia"></i>
      <div class="text">{{text}}</div>
    </div>
    <div @click="btnEventClick" class="del-btn btn-position icon-theme"
         v-else-if=" typeStr==='del'">
      <i class="icon-i iconfontBtn icon-guanbi"></i>
      <span class="text">{{text}}</span>
    </div>
    <div @click="btnEventClick" class="export-btn btn-position icon-theme"
         v-else-if=" typeStr==='export'">
      <span class="icon-i iconfontBtn icon-daochu"></span>
      <span class="text">{{text}}</span>
    </div>
    <div @click="btnEventClick" class="search-btn btn-position icon-theme"
         v-else-if=" typeStr==='search'">
      <span class="icon-i iconfontBtn icon-sousuo"></span>
      <span class="text">{{text}}</span>
    </div>
    <div @click="btnEventClick" class="refresh-btn btn-position icon-theme"
         v-else-if=" typeStr==='refresh'">
      <span class="icon-i iconfontBtn icon-shuaxin"></span>
      <span class="text">{{text}}</span>
    </div>
    <div @click="btnEventClick" class="update-btn btn-position icon-theme"
         v-else-if=" typeStr==='update'">
      <span class="icon-i iconfontBtn icon-xiugai"></span>
      <span class="text">{{text}}</span>
    </div>
    <div @click="btnEventClick" class="reset-btn btn-position icon-theme"
         v-else-if=" typeStr==='reset'">
      <span class="icon-i" :class="reset_icon"></span>
      <span class="text">{{text}}</span>
    </div>
    <div @click="btnEventClick" class=" default-btn btn-position btn-theme"
         v-else>
      {{text}}
    </div>
  </div>
</template>

<script>
export default {
  name: 'ButtonTem',
  props: {
    typeStr: {
      type: String,
      default: () => 'default'
    },
    text: {
      type: String,
      default: () => 'Button'
    },
    reset_icon: {
      type: String,
      default: () => 'iconfontBtn icon-zhongzhi'
    }
  },
  methods: {
    /**
     * @event: 点击
     */
    btnEventClick() {
      this.$emit('click')
    }
  },
};
</script>
