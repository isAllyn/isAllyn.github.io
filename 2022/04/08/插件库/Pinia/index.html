pinia 学习笔记

2022年04月08日
插件库
留言

install

npm i -S pinia

use

import { createPinia } from 'pinia'

app.use(createPinia())

difineStore

import { defineStore } from 'pinia'

export const test_store = defineStore('test_store', {
  state: () => {
    return {
      name: 'Boring ghost!',
    }
  },
  getters: {
    new_name: (state) => 'new ' + state.name,
  },
  actions: {
    refresh() {
      this.name = 'refresh Boring ghost!'
    },
  },
})

at component

import { test_store } from '@/store/store_test'
const store = test_store()
    <div>this is store state value:{{ store.name }}</div>

set state

直接解构会失去响应式

const {name} = test_store() //At question! this is error!
<!-- bottom is no problem -->
import {storeToRefs} from "pinia"
const {name} = storeToRefs(test_store())

  • 方式一

    import { test_store } from '@/store/store_test'
const store = test_store()
store.name='new value'

  • 方式二
    批量修改

    store.$patch({
	name:'new value',
	key2:"new value"
})

  • 方式三(推荐)

    store.$patch((state)=>{
	state.name="new value"
})

pinia Actions

支持同步跟异步

import { test_store } from '@/store/store_test'
const store = test_store()
store.fn()

API

  • $reset

    将state回复成初始状态

    import { test_store } from '@/store/store_test'
const store = test_store()
store.$reset() //直接调用不需要传参

  • $subscribe

    state发生变化就会触发

    store.$subscribe((args,state)=>{},{
	detached:true,//true:组件销毁后任然监听
	deep:true
})

  • $onAction

    调用actions就会被触发

    store.$onAction((args)=>{},true)//true:组件销毁后任然监听
