v3 use

npm i -S vue-i18n@next

config

const i18n=createI18n({
  locale:"",
  fallbackLocale?: 'zh-CN',//没有英文的时候默认中文语言
  locale: 'zh-CN',    // 语言标识
  globalInjection: true, //使用全局$t()函数

})
export default i18n
app.use(i18n) the main.js
use the template:
<button>{{ $t("title") }}</button>

  let i18n = useI18n();
  i18n.locale.value = e.target.innerText;

Function

const getCurrentLanguge = () => {
  const win_lang = navigator.language//浏览器使用的语言
  const current_lang = win_lang.indexOf('zh') !== -1 ? 'zh' : 'en'
  localStorage.setItem('lang', current_lang)
  return current_lang
}

pinia 学习笔记

2022年04月08日
插件库
留言

install

npm i -S pinia

use

import { createPinia } from 'pinia'

app.use(createPinia())

difineStore

import { defineStore } from 'pinia'

export const test_store = defineStore('test_store', {
  state: () => {
    return {
      name: 'Boring ghost!',
    }
  },
  getters: {
    new_name: (state) => 'new ' + state.name,
  },
  actions: {
    refresh() {
      this.name = 'refresh Boring ghost!'
    },
  },
})

at component

import { test_store } from '@/store/store_test'
const store = test_store()
    <div>this is store state value:{{ store.name }}</div>

set state

直接解构会失去响应式

const {name} = test_store() //At question! this is error!
<!-- bottom is no problem -->
import {storeToRefs} from "pinia"
const {name} = storeToRefs(test_store())

  • 方式一

    import { test_store } from '@/store/store_test'
const store = test_store()
store.name='new value'

  • 方式二
    批量修改

    store.$patch({
	name:'new value',
	key2:"new value"
})

  • 方式三(推荐)

    store.$patch((state)=>{
	state.name="new value"
})

pinia Actions

支持同步跟异步

import { test_store } from '@/store/store_test'
const store = test_store()
store.fn()

API

  • $reset

    将state回复成初始状态

    import { test_store } from '@/store/store_test'
const store = test_store()
store.$reset() //直接调用不需要传参

  • $subscribe

    state发生变化就会触发

    store.$subscribe((args,state)=>{},{
	detached:true,//true:组件销毁后任然监听
	deep:true
})

  • $onAction

    调用actions就会被触发

    store.$onAction((args)=>{},true)//true:组件销毁后任然监听

安装

相比webpack

  • vite 直接type=’module’ 引入就预览,而 webpack 打一次包后再预览
  • vite HMR热更新,只刷新更新部分的内容,非所有内容
  • vite Rollip 插件打包

使用ts

  1. 选择 vue-ts 模板时自带 typescript, 只需要npm i

配置 resolve.alias

path会报2307,根据提示install @types/node 即可
vite.config.ts

import { resolve } from 'path'
export default defineConfig({
	resolve:{
		alias:{
			'@':resolve(__dirname,'src')
		}
	}
})

tsconfig.json add+

"baseUrl": ".",
"paths": {
   "@/*": ["src/*"],
   "@page/*": ["src/page/*"],
   "@img/*": ["src/asstes/img/*"]
}

  • 压缩插件

    压缩静态资源较大的文件,
    一些js，css，html，以及静态资源。gzip是用算法压缩文件，节省流量。等文件到达浏览器后，浏览器会解压文件，解压出的源文件与gzip之前的大小是没变化的。但是传输过程的流量消耗和带宽节省了。
    npm i -D vite-plugin-compression

    
import viteCompression from 'vite-plugin-compression'

viteCompression({
    verbose: true,
    disable: false,
    threshold: 10240,
    algorithm: 'gzip',
    ext: '.gz',
  })

      compress: {
        drop_console: true,
        drop_debugger: true
      }
  }
}

const { resolve } = require('path')

const CleanWebpackPlugin = require('clean-webpack-plugin')

  entry: './js/index.js',//入口文件
  devServer: {//开发服务的配置,只打包到内存,不做实际打包,作用于开发时实时预览

    index: 'index.html',//索引文件的文件名,默认就是index.html,配置一下应该启动会快一点..
    publicPath: '/',//静态资源访问路径 默认就是/ => 比如: ip/build.js (文档有说明)
    hot: true,//开启HMR
    overlay: true,//代码出错开启全屏错误提示
    watchContentBase: true,// 监视contentBase目录下所有文件
    watchOptions: {
      ignored: /node_modules/ //忽略文件
    },
    clientLogLevel: 'none',//不要显示启动服务的日志信息 
    // quiet: true,//除了初始启动信息之外的任何内容都不会被打印到控制台 TODO :小项目没必要关闭,还可以检查一下打包的信息   
  },
  output: {//打包出口配置
    filename: 'js/main.js',//出口文件的文件名
    path: resolve(__dirname, 'dist'),//所有打包后的文件放入dist文件夹下

  },

    rules: [

                }
              }
            ]
          },

            test: /\.less$/i,// TODO :less不要跟css文件混搭使用
            use: [
              {
                loader: MiniCssExtractPlugin.loader,
                options: {
                  publicPath: '../'
                }
              },
              'css-loader',
              'less-loader',//编译less文件

                  ident: 'postcss',

                  ]
                }
              }

              // 'style-loader',

                  publicPath: '../' // TODO :因为css单独提取是配置了提取到css文件夹下,也就是多出了一层css,所以这里也要相应的回退一层
                }

                  ]
                }
              }
            ]
          },
          {
            test: /\.ts$/i,
            loader: 'ts-loader',
          },
          {
            test: /\.(png|jpg|gif)$/i,
            loader: 'url-loader',// 处理图片路径 TODO :无法处理html文件的图片,需要搭配html-loader
            options: {
              limit: 0,//小于0k编译成base64 优点:减少服务请求(服务压力) 缺点:文件变大,加载变慢
              esModule: false,// TODO :html-loader用的commonjs写法引入图片,需要将url-loader esmodule写法关闭,否则html无法解析
